Configuración

Conecta Google Sheets para guardar los datos en la nube y compartir con todos los profesores.

⚠️ Sin Google Sheets los datos se guardan solo en este navegador. Configura la integración para que todos los profesores puedan registrar asistencia y los datos queden centralizados.

1
Código para Google Apps Script

Abre tu Google Sheet → Extensiones → Apps Script. Borra el contenido que aparece y pega este código:

function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); var sn = d.tipo === 'falta' ? 'Faltas' : 'Asistencia'; var sh = ss.getSheetByName(sn) || ss.insertSheet(sn); if (sh.getLastRow() === 0) sh.appendRow(d.tipo==='falta' ? ['Fecha','Hora','Estudiante','ID','Programa','Clase','Profesor','Faltas acum.'] : ['Fecha','Hora','Estudiante','ID','Programa','Clase','Profesor']); sh.appendRow(d.tipo==='falta' ? [d.fecha,d.hora,d.estudiante,d.id,d.programa||'',d.clase,d.profesor,d.total] : [d.fecha,d.hora,d.estudiante,d.id,d.programa||'',d.clase,d.profesor]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(){ return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
