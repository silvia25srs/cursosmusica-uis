function doGet(e) {
  var p = (e && e.parameter) ? e.parameter : {};
  var action = p.action || 'ping';
  var result;
  try {
    if      (action === 'ping')                result = { ok:true, msg:'Conectado OK v2026-6' };
    else if (action === 'registrar_asistencia') result = registrarAsistencia(p);
    else if (action === 'registrar_falta')      result = registrarFalta(p);
    else if (action === 'guardar_evidencia')    result = guardarEvidencia(p);
    else if (action === 'guardar_logistica')    result = guardarLogistica(p);
    else if (action === 'asistencia')           result = getData('Asistencia', p.profesor);
    else if (action === 'faltas')               result = getData('Faltas', p.profesor);
    else if (action === 'evidencias')           result = getData('Evidencias', p.profesor);
    else if (action === 'logistica')            result = getData('Logistica', p.profesor);
    else result = { ok:false, msg:'Accion no reconocida: ' + action };
  } catch(err) { result = { ok:false, error: err.toString() }; }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    return doGet({ parameter: data });
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ ok:false, error:err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

function hoy()  { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy'); }
function ahora(){ return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'HH:mm:ss'); }

function getOrCreateSheet(name, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.appendRow(headers);
    sh.getRange(1,1,1,headers.length).setFontWeight('bold').setBackground('#31511E').setFontColor('#ffffff');
    sh.setFrozenRows(1);
  }
  return sh;
}

function getData(sheetName, filterProfesor) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(sheetName);
  if (!sh || sh.getLastRow() < 2) return { ok:true, data:[] };
  var rows = sh.getDataRange().getValues();
  var headers = rows[0];
  var data = rows.slice(1).map(function(r){
    var obj = {}; headers.forEach(function(h,i){ obj[h] = r[i]; }); return obj;
  });
  if (filterProfesor) data = data.filter(function(r){ return r['Profesor'] === filterProfesor; });
  return { ok:true, data:data };
}

function registrarAsistencia(p) {
  getOrCreateSheet('Asistencia',['ID','Fecha','Hora','Estudiante','Clase','Programa','Profesor'])
    .appendRow([p.id||'', p.fecha||hoy(), p.hora||ahora(), p.estudiante||'', p.clase||'', p.programa||'', p.profesor||'']);
  return { ok:true };
}

function registrarFalta(p) {
  getOrCreateSheet('Faltas',['Fecha','Estudiante','Clase','Programa','Hora','Profesor'])
    .appendRow([p.fecha||hoy(), p.estudiante||'', p.clase||'', p.programa||'', p.hora||'', p.profesor||'']);
  return { ok:true };
}

function guardarEvidencia(p) {
  getOrCreateSheet('Evidencias',['Fecha','Hora','Tipo','Salon','Observaciones','Profesor','Archivo'])
    .appendRow([p.fecha||hoy(), p.hora||ahora(), p.tipo||'', p.salon||'', p.obs||'', p.profesor||'', p.fileName||'']);
  return { ok:true };
}

function guardarLogistica(p) {
  var sh = getOrCreateSheet('Logistica',['Fecha','Clase','Elementos','Profesor']);
  if (sh.getLastRow() > 1) {
    var rows = sh.getDataRange().getValues();
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][1] === p.clase && rows[i][3] === p.profesor) {
        sh.getRange(i+1,1,1,4).setValues([[hoy(),p.clase||'',p.items||'',p.profesor||'']]);
        return { ok:true };
      }
    }
  }
  sh.appendRow([hoy(), p.clase||'', p.items||'', p.profesor||'']);
  return { ok:true };
}
