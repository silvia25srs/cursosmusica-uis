Configuración

Conecta Google Sheets para guardar los datos en la nube y compartir con todos los profesores.

⚠️ Sin Google Sheets los datos se guardan solo en este navegador. Configura la integración para que todos los profesores puedan registrar asistencia y los datos queden centralizados.

1
Código para Google Apps Script

Abre tu Google Sheet → Extensiones → Apps Script. Borra el contenido que aparece y pega este código:

function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); var sn, sh, headers, row; if (d.tipo === 'evidencia') { sn = 'Evidencias'; sh = ss.getSheetByName(sn) || ss.insertSheet(sn); headers = ['Fecha','Hora','Profesor','Clase','Día','Espacio','Instrumentos','Observaciones','Archivos']; row = [d.fecha,d.hora,d.profesor,d.clase,d.dia,d.espacio,d.instrumentos,d.observaciones||'',d.archivos||'']; } else if (d.tipo === 'falta') { sn = 'Faltas'; sh = ss.getSheetByName(sn) || ss.insertSheet(sn); headers = ['Fecha','Hora','Estudiante','ID','Programa','Clase','Profesor','Faltas acum.']; row = [d.fecha,d.hora,d.estudiante,d.id,d.programa||'',d.clase,d.profesor,d.total]; } else { sn = 'Asistencia'; sh = ss.getSheetByName(sn) || ss.insertSheet(sn); headers = ['Fecha','Hora','Estudiante','ID','Programa','Clase','Profesor']; row = [d.fecha,d.hora,d.estudiante,d.id,d.programa||'',d.clase,d.profesor]; } if (sh.getLastRow() === 0) sh.appendRow(headers); sh.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(){ return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
