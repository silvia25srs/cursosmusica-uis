📊 Google Sheets — Asistencia, Faltas, Evidencias y Logística
Toda la información de la app se guarda automáticamente en tu hoja de cálculo de Google.
📋 Pasos para configurar:
  1. Abre Google Sheets con tu cuenta UIS y crea una hoja nueva
  2. Nómbrala: Asistencia Cursos Música UIS 2026-1
  3. En el menú ve a Extensiones → Apps Script
  4. Borra el código de ejemplo y pega el script completo de abajo
  5. Guarda con Ctrl + S y ponle nombre: MusicUIS Web App
  6. Clic en Implementar → Nueva implementación
  7. En "Tipo" selecciona Aplicación web
  8. "Ejecutar como" → Yo (tu cuenta de Google)
  9. "Quién puede acceder" → Cualquier persona
  10. Clic en Implementar → Autoriza los permisos → Copia la URL
  11. Pega la URL en el campo de abajo y guarda
📄 Script Web App — copia y pega en Apps Script:
// ============================================================
// CURSOS DE MÚSICA UIS — Google Apps Script Web App
// Versión: 2026-1

// INSTRUCCIONES DE INSTALACIÓN:
// 1. Abre Google Sheets en la hoja "Asistencia Cursos Música UIS 2026-1"
// 2. Ve a Extensiones → Apps Script
// 3. Borra el contenido existente y pega TODO este código
// 4. Guarda (Ctrl+S)
// 5. Clic en "Implementar" → "Nueva implementación"
// 6. Tipo: Aplicación web
// 7. Ejecutar como: Yo (tu cuenta de Google)
// 8. Quién puede acceder: Cualquier persona
// 9. Clic en "Implementar" y copia la URL generada

const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

// ---- HOJAS ----
const HOJA_ASISTENCIA = 'Asistencia';
const HOJA_FALTAS     = 'Faltas';
const HOJA_EVIDENCIAS = 'Evidencias';
const HOJA_LOGISTICA  = 'Logistica';

// ---- CORS ----
function setCORSHeaders(output) {
  return output
    .setHeader('Access-Control-Allow-Origin', '*')
    .setHeader('Access-Control-Allow-Methods', 'GET, POST')
    .setHeader('Access-Control-Allow-Headers', 'Content-Type');
}

// ---- ENTRY POINTS ----
function doGet(e) {
  const action = e.parameter.action || '';
  let result;
  try {
    if (action === 'ping')          result = { ok: true, msg: 'Conectado correctamente ✅' };
    else if (action === 'asistencia') result = getAsistencia(e.parameter);
    else if (action === 'faltas')     result = getFaltas(e.parameter);
    else if (action === 'evidencias') result = getEvidencias(e.parameter);
    else if (action === 'logistica')  result = getLogistica(e.parameter);
    else result = { ok: false, msg: 'Acción no reconocida: ' + action };
  } catch(err) {
    result = { ok: false, error: err.message };
  }
  return setCORSHeaders(
    ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON)
  );
}

function doPost(e) {
  let data, action, result;
  try {
    data   = JSON.parse(e.postData.contents);
    action = data.action || '';
    if (action === 'registrar_asistencia') result = registrarAsistencia(data);
    else if (action === 'registrar_falta')  result = registrarFalta(data);
    else if (action === 'guardar_evidencia') result = guardarEvidencia(data);
    else if (action === 'guardar_logistica') result = guardarLogistica(data);
    else result = { ok: false, msg: 'Acción no reconocida: ' + action };
  } catch(err) {
    result = { ok: false, error: err.message };
  }
  return setCORSHeaders(
    ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON)
  );
}

// ASISTENCIA
// ============================================================
function registrarAsistencia(data) {
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet   = ss.getSheetByName(HOJA_ASISTENCIA);
  if (!sheet) {
    sheet = ss.insertSheet(HOJA_ASISTENCIA);
    sheet.appendRow(['ID','Fecha','Hora','Estudiante','Clase','Programa','Hora Clase','Profesor','Registrado por']);
    sheet.getRange(1,1,1,9).setFontWeight('bold').setBackground('#31511E').setFontColor('#ffffff');
  }
  sheet.appendRow([
    data.id       || '',
    data.fecha    || new Date().toLocaleDateString('es-CO'),
    data.hora     || new Date().toLocaleTimeString('es-CO'),
    data.estudiante || '',
    data.clase    || '',
    data.programa || '',
    data.horaClase || '',
    data.profesor || '',
    data.registradoPor || '',
  ]);
  return { ok: true, msg: 'Asistencia registrada' };
}

function getAsistencia(params) {
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(HOJA_ASISTENCIA);
  if (!sheet) return { ok: true, data: [] };
  const rows = sheet.getDataRange().getValues();
  const headers = rows[0];
  const data = rows.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = r[i]);
    return obj;
  });
  // Filter by profesor if provided
  if (params.profesor) {
    return { ok: true, data: data.filter(r => r['Profesor'] === params.profesor) };
  }
  return { ok: true, data };
}

function guardarLogistica(data) {
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet   = ss.getSheetByName(HOJA_LOGISTICA);
  if (!sheet) {
    sheet = ss.insertSheet(HOJA_LOGISTICA);
    sheet.appendRow(['Fecha','Clase','Elementos','Profesor']);
    sheet.getRange(1,1,1,4).setFontWeight('bold').setBackground('#31511E').setFontColor('#ffffff');
  }
  // Update existing or append
  const rows = sheet.getDataRange().getValues();
  const existing = rows.findIndex(r => r[1] === data.clase && r[3] === data.profesor);
  if (existing > 0) {
    sheet.getRange(existing + 1, 1, 1, 4).setValues([[
      new Date().toLocaleDateString('es-CO'), data.clase, data.items, data.profesor
    ]]);
  } else {
    sheet.appendRow([
      new Date().toLocaleDateString('es-CO'),
      data.clase || '', data.items || '', data.profesor || '',
    ]);
  }
  return { ok: true, msg: 'Logística guardada' };
}

function getLogistica(params) {

}
