var DRIVE_FOLDER_ID = '1pXADsK5eLk3SHcaqIe9VaY_VhysexLA8';

function doGet(e) {
  var p = (e && e.parameter) ? e.parameter : {};
  try { return out(handle(p)); }
  catch(err) { return out({ ok:false, error:err.toString() }); }
}

function doPost(e) {
  var p = {};
  try {
    if (e.postData && e.postData.contents)
      p = JSON.parse(e.postData.contents);
    else
      p = e.parameter || {};
  } catch(x) { p = e.parameter || {}; }
  try { return out(handle(p)); }
  catch(err) { return out({ ok:false, error:err.toString() }); }
}

function out(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function handle(p) {
  var a = p.action || 'ping';
  if (a === 'ping')               return { ok:true, msg:'Conectado v2026-7' };
  if (a === 'registrar_asistencia') return registrarAsistencia(p);
  if (a === 'registrar_falta')      return registrarFalta(p);
  if (a === 'guardar_evidencia')    return guardarEvidencia(p);
  if (a === 'guardar_logistica')    return guardarLogistica(p);
  if (a === 'asistencia')  return getData('Asistencia', p.profesor);
  if (a === 'faltas')      return getData('Faltas',     p.profesor);
  if (a === 'evidencias')  return getData('Evidencias', p.profesor);
  if (a === 'logistica')   return getData('Logistica',  p.profesor);
  return { ok:false, msg:'Accion no reconocida: ' + a };
}

function hoy()  { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy'); }
function ahora(){ return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'HH:mm:ss'); }

function sheet(name, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.appendRow(headers);
    sh.getRange(1,1,1,headers.length)
      .setFontWeight('bold').setBackground('#31511E').setFontColor('#ffffff');
    sh.setFrozenRows(1);
  }
  return sh;
}

function getData(name, prof) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh || sh.getLastRow() < 2) return { ok:true, data:[] };
  var rows = sh.getDataRange().getValues();
  var heads = rows[0];
  var data = rows.slice(1).map(function(r){
    var o={}; heads.forEach(function(h,i){ o[h]=r[i]; }); return o;
  });
  if (prof) data = data.filter(function(r){ return r['Profesor']===prof; });
  return { ok:true, data:data };
}

function registrarAsistencia(p) {
  sheet('Asistencia',['ID','Fecha','Hora','Estudiante','Clase','Programa','Profesor'])
    .appendRow([p.id||'',p.fecha||hoy(),p.hora||ahora(),
                p.estudiante||'',p.clase||'',p.programa||'',p.profesor||'']);
  return { ok:true };
}

function registrarFalta(p) {
  sheet('Faltas',['Fecha','Estudiante','Clase','Programa','Hora','Profesor'])
    .appendRow([p.fecha||hoy(),p.estudiante||'',p.clase||'',
                p.programa||'',p.hora||'',p.profesor||'']);
  return { ok:true };
}

// guardarEvidencia: guarda metadatos Y sube el archivo a Drive en un solo paso
function guardarEvidencia(p) {
  var driveUrl = '';

  // Si viene el archivo en base64, subirlo a Drive ahora
  if (p.fileBase64 && p.fileBase64.length > 10) {
    try {
      var folder = DriveApp.getFolderById(DRIVE_FOLDER_ID);
      var bytes  = Utilities.base64Decode(p.fileBase64);
      var mime   = p.fileMime || 'image/jpeg';
      var name   = p.fileName || ('foto_' + hoy() + '.jpg');
      var blob   = Utilities.newBlob(bytes, mime, name);
      var file   = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      driveUrl = file.getUrl();
    } catch(e) {
      driveUrl = 'Error-Drive: ' + e.message;
    }
  }

  sheet('Evidencias',['Fecha','Hora','Tipo','Clase','Salon','Observaciones','Profesor','Ventanas Cerradas','Sillas Ordenadas','Ventilador/Aire Apagado'])
    .appendRow([p.fecha||hoy(), p.hora||ahora(), p.tipo||'',
                p.clase||'', p.salon||'', p.obs||'', p.profesor||'',
                p.ventanas||'', p.sillas||'', p.ventilador||'']);
  return { ok:true };
}

function guardarLogistica(p) {
  sheet('Logistica',['Fecha Envío','Sábado','Clase','Jornada','Salón','Hora','Elementos','Profesor'])
    .appendRow([hoy(),p.fecha||'',p.clase||'',p.jornada||'',
                p.salon||'',p.hora||'',p.items||'',p.profesor||'']);
  return { ok:true };
}
